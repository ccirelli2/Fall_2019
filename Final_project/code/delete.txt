SELECT 	full_name, 
	root_name, 
	writer_id,
	dob, 
	education, 
	gender, 
	country_origin, 
	native_language, 
	writing_hand 

FROM 	file_mapping

JOIN	author_image_mapping ON file_mapping.root_name = author_image_mapping.form_id

JOIN	author_info ON author_image_mapping.writer_id = author_info.name
